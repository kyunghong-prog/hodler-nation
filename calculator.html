    <script>

        // Get elements

        const ageInput = document.getElementById('age');

        const savingsInput = document.getElementById('savings');

        const monthlyInput = document.getElementById('monthly');

        

        const ageValue = document.getElementById('age-value');

        const savingsValue = document.getElementById('savings-value');

        const monthlyValue = document.getElementById('monthly-value');

        

        // Format currency

        function formatCurrency(value) {

            return new Intl.NumberFormat('en-US', {

                style: 'currency',

                currency: 'USD',

                minimumFractionDigits: 0,

                maximumFractionDigits: 0,

            }).format(value);

        }

        

        // Get allocation based on age

        function getAllocation(age) {

            if (age < 30) return { spy: 20, qqq: 80, return: 0.131 };

            if (age < 40) return { spy: 30, qqq: 70, return: 0.126 };

            if (age < 50) return { spy: 50, qqq: 50, return: 0.120 };

            if (age < 60) return { spy: 60, qqq: 40, return: 0.117 };

            return { spy: 70, qqq: 30, return: 0.114 };

        }

        

        // Calculate results

        function calculate() {

            const age = parseInt(ageInput.value);

            const currentSavings = parseInt(savingsInput.value);

            const monthlyContribution = parseInt(monthlyInput.value);

            

            // Update display values

            ageValue.textContent = age;

            savingsValue.textContent = formatCurrency(currentSavings);

            monthlyValue.textContent = formatCurrency(monthlyContribution);

            

            // Get allocation

            const allocation = getAllocation(age);

            document.getElementById('allocation').textContent = 

                `${allocation.spy}% SPY / ${allocation.qqq}% QQQ`;

            document.getElementById('expected-return').textContent = 

                `Expected Return: ${(allocation.return * 100).toFixed(1)}% annually`;

            

            // Calculate

            const retirementAge = 65;

            const yearsToRetirement = retirementAge - age;

            const monthlyRate = allocation.return / 12;

            const months = yearsToRetirement * 12;

            

            const futureOfSavings = currentSavings * Math.pow(1 + monthlyRate, months);

            const futureOfContributions = monthlyContribution * 

                ((Math.pow(1 + monthlyRate, months) - 1) / monthlyRate);

            

            const finalValue = Math.round(futureOfSavings + futureOfContributions);

            const totalContributed = Math.round(currentSavings + (monthlyContribution * months));

            const totalGains = finalValue - totalContributed;

            const reachesGoal = finalValue >= 1000000;

            

            // Update results

            document.getElementById('final-value').textContent = formatCurrency(finalValue);

            document.getElementById('gains-value').textContent = formatCurrency(totalGains);

            document.getElementById('contributed-value').textContent = formatCurrency(totalContributed);

            document.getElementById('years-detail').textContent = `Over ${yearsToRetirement} years`;

            

            // Update goal status

            const finalCard = document.getElementById('final-card');

            if (reachesGoal) {

                finalCard.className = 'result-card success';

                document.getElementById('goal-status').textContent = '✅ You will reach $1M!';

            } else {

                finalCard.className = 'result-card warning';

                document.getElementById('goal-status').textContent = 

                    `⚠️ Short by ${formatCurrency(1000000 - finalValue)}`;

            }

        }

        

        // Add event listeners

        ageInput.addEventListener('input', calculate);

        savingsInput.addEventListener('input', calculate);

        monthlyInput.addEventListener('input', calculate);

        

        // Initial calculation

        calculate();

    </script>
